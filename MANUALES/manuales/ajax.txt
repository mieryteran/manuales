	function addEvent(elemento,evento,funcion,capture)
	{
		if(elemento.attachEvent){
		elemento.attachEvent('on'+evento,funcion);
		return true;
		}else if(elemento.addEventListener){
		elemento.addEventListener(evento,funcion,capture);
		return true;
		}else{
		return false;
		}
	}
